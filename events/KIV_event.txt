add_namespace = KIV
add_namespace = KIV_incomplete

country_event = {
	id = KIV.1
	title = KIV.1.title
	desc = KIV.1.desc
	is_triggered_only = yes
	hidden = yes
	immediate = {
		hidden_effect = {
			KIV = {
				load_focus_tree = KIV_general
				KIV_GSC_president = {
					remove_unit_leader = yes
				}
				remove_country_leader_role = {
					character = KIV_GSC_president
					ideology = sub_Techno_Bureaucracy
				}
				promote_character = {
					character = KIV_qishenlin
					ideology = sub_Techno_Bureaucracy
				}
				set_politics = {
					ruling_party = democratic
					elections_allowed = no
				}
				remove_ideas = KIV_idea_federal_student_union
				add_timed_idea = { idea = KIV_chaos_congress days = 365 }

				add_to_array = { IRIS_Historical_leaders_Array = token:KIV_qishenlin } #七神 凛
				add_to_array = { IRIS_Historical_leaders_2_Array = token:KIV_oki_aoi } #扇喜 葵
				add_to_array = { IRIS_Historical_leaders_voting_rate_Array = 1005 }
				add_to_array = { GetNameIRISHistoricalDate = global.year }
			}
		}
	}
	option = {
		name = KIV.1.a
	}
}

country_event =  {
	id = KIV_incomplete.1
	title = KIV_incomplete.1.t
	desc = KIV_incomplete.1.desc
	hidden = yes
	trigger = {
		tag = KIV
		KIV = {
			NOT = {
				has_government = fascism
			}
			has_civil_war = no
			fascism > 0.5
		}
	}
	immediate = {
		KIV = {
			set_politics = { ruling_party = fascism }
			add_to_array = { IRIS_Historical_leaders_Array = token:KIV_hanuma_makoto } #羽沼 真琴
			add_to_array = { IRIS_Historical_leaders_2_Array = token:KIV_motomiya_chiaki } #元宫 千明
			add_to_array = { IRIS_Historical_leaders_voting_rate_Array = 1005 }
			add_to_array = { GetNameIRISHistoricalDate = global.year }
		}
	}
	fire_only_once = no

	option = {
		name = KIV_incomplete.1.a

	}

}
