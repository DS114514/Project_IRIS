ANZ_frontline_control_1 = {
	allowed = {
		original_tag = ANZ
	}
	enable = {
		always = yes
	}
	abort = {
		always = no
	}

	ai_strategy = {
		type = front_control

		state_trigger = {
			NOT = {
				is_owned_by = FROM.FROM
			}
		}
		
		ordertype = front
		execution_type = careful
	}
}

	allowed = {

	}
	enable = {
		set_temp_variable = { days = global.num_days }
		modulo_temp_variable = { days = 90 }
		check_variable = { days < 30 }
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = front_control

		state_trigger = {
			NOT = {
				is_owned_by = FROM.FROM
			}
		}
		
		ordertype = front
		execute_order = no
	}
}
