on_actions = {
    on_startup = {
        effect = {
            CTC = {
                set_variable = {CTC_ELN_influence = 20}
                give_resource_rights = {
                    receiver = ELN
                    state = 143
                    resources = { oil }
                }
                give_resource_rights = {
                    receiver = ELN
                    state = 132
                    resources = { oil }
                }
            }
        }
    }
    on_weekly_CTC = {
        effect = {
            if = {
                limit = {
                    CTC = {
                        has_completed_focus = CTC_petrol_station_of_europa
                        has_capitulated = no
                    }
                }
                every_country = {
                    limit = {
                        OR = {
                            is_in_faction_with = CTC
                            is_puppet_of = CTC
                            CTC = {is_puppet_of = PREV}
                        }
                    }
                    add_ideas = CTC_oil_supply
                }
            }
        }
    }
    on_capitulation_immediate = {
        effect = {
            if = {
                limit = {
                    ROOT = {
                        tag = CTC
                    }
                    FROM = {
                        OR = {
                            tag = DOC
                            tag = OCF
                            tag = ELN
                            is_in_faction_with = DOC
                            is_in_faction_with = OCF
                            is_in_faction_with = ELN
                        }
                    }
                }
                every_country = {
                    limit = {
                        is_in_faction_with = FROM
                    }
                    white_peace = {
                        tag = ROOT
                    }
                }
                FROM = {
                    set_autonomy = {
                        target = ROOT
                        autonomous_state = autonomy_puppet
                    }
                    every_controlled_state = {
                        limit = {
                            is_core_of = ROOT
                            NOT = {is_core_of = FROM}
                        }
                        transfer_state_to = ROOT
                    }
                }
            }
        }
    }
}
