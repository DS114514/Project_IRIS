on_actions = {
	on_startup = {
		effect = {
            # DOC诗旎开局设定
			if = {
				limit = {
					has_game_rule = {
						rule = DOC_ai_behavior
						option = DEFAULT
					}
				}
				#
			}
			if = {
				limit = {
					has_game_rule = {
						rule = DOC_ai_behavior
						option = DOC_RULE_OPTION_RANDOM
					}
				}
				set_global_flag = DOC_BOSS_FLAG
			}
			if = {
				limit = {
					has_game_rule = {
						rule = DOC_ai_behavior
						option = DOC_RULE_OPTION_ALLOWED
					}
				}
				set_global_flag = DOC_BOSS_FLAG_100
			}
			
			# MAG神滨开局设定
			if = {
				limit = {
					has_game_rule = {
						rule = MAG_ai_behavior
						option = DEFAULT
					}
				}
				MAG = {
					set_country_flag = MAG_AR_JZCX
				}
			}
			if = {
				limit = {
					has_game_rule = {
						rule = MAG_ai_behavior
						option = MAG_RULE_OPTION_RANDOM
					}
				}
				#
			}
			if = {
				limit = {
					has_game_rule = {
						rule = MAG_ai_behavior
						option = MAG_RULE_OPTION_ALLOWED
					}
				}
				MAG = {
					set_country_flag = MAG_AR_RTLB
				}
			}
		}
	}

	# 圆环之理教会与进步理事会瓜分博闻
	on_capitulation = {
		effect = {
			if = {
				limit = {
					ROOT = BOW
					FROM = PCC
					MAG = {
						has_completed_focus = MAG_AD_gmjzmld
						is_in_faction_with = PCC
					}
				}
				every_country = {
					news_event = {
						id = MAG_A.49
					}
				}
			}
		}
	}

	#停止 强制民众祷告
	on_weekly_MAG = {
		if = {
			limit = {
				power_balance_value = {
					id = MAG_AN_BOP_dyhzldxycd
					value > 0.35
				}
			}
			remove_power_balance_modifier = {
				id = MAG_AN_BOP_dyhzldxycd
				modifier = MAG_AN_qzmzdg_modifier_weekly
			} 
		}
	}
}
