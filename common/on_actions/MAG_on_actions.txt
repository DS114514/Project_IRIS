on_actions = {
	on_startup = {
		effect = {
            # DOC诗旎开局设定
			if = {
				limit = {
					has_game_rule = {
						rule = DOC_ai_behavior
						option = DEFAULT
					}
				}
				#
			}
			if = {
				limit = {
					has_game_rule = {
						rule = DOC_ai_behavior
						option = DOC_RULE_OPTION_RANDOM
					}
				}
				set_global_flag = DOC_BOSS_FLAG
			}
			if = {
				limit = {
					has_game_rule = {
						rule = DOC_ai_behavior
						option = DOC_RULE_OPTION_ALLOWED
					}
				}
				set_global_flag = DOC_BOSS_FLAG_100
			}
			
			# MAG神滨开局设定
			if = {
				limit = {
					has_game_rule = {
						rule = MAG_ai_behavior
						option = DEFAULT
					}
				}
				MAG = {
					set_country_flag = MAG_AR_JZCX
				}
			}
			if = {
				limit = {
					has_game_rule = {
						rule = MAG_ai_behavior
						option = MAG_RULE_OPTION_RANDOM
					}
				}
				#
			}
			if = {
				limit = {
					has_game_rule = {
						rule = MAG_ai_behavior
						option = MAG_RULE_OPTION_ALLOWED
					}
				}
				MAG = {
					set_country_flag = MAG_AR_RTLB
				}
			}
		}
	}
}
