OCF_obtain_banknotes = {

  add_to_variable = { OCF_obtain_banknotes_variable_markup = tax_impact }

  multiply_variable = { tax_impact_100 = 0.01 }#加成数值

#花费资金
OCF_spending_funds = {
  set_variable = { OCF_spending_funds_variable = 2 }
  if = {
    limit = { has_war = yes }
    set_variable = { OCF_spending_funds_variable_has_war = 2.5 }
    subtract_from_variable = { OCF_spending_funds_variable_has_war = ROOT.has_war_support }#获取战争支持
    set_variable = { OCF_number_military_personnel = ROOT.deployed_army_manpower_k }
    add_to_variable = { OCF_number_military_personnel = ROOT.deployed_navy_manpower_k }
    add_to_variable = { OCF_number_military_personnel = ROOT.deployed_airforce_manpower_k }
    divide_variable = { OCF_number_military_personnel = 750 }#军队花费
    add_to_variable = { OCF_number_military_personnel = 1 }
    
    multiply_variable = { OCF_spending_funds_variable = OCF_spending_funds_variable_has_war }
    multiply_variable = { OCF_spending_funds_variable = OCF_number_military_personnel }#减去军队花费
  }else = {
    set_variable = { OCF_spending_funds_variable_has_war = 2.5 }
    subtract_from_variable = { OCF_spending_funds_variable_has_war = ROOT.has_war_support }#获取战争支持
    set_variable = { OCF_number_military_personnel = ROOT.deployed_army_manpower_k }
    add_to_variable = { OCF_number_military_personnel = ROOT.deployed_navy_manpower_k }
    add_to_variable = { OCF_number_military_personnel = ROOT.deployed_airforce_manpower_k }
    divide_variable = { OCF_number_military_personnel = 1500 }#军队花费
    add_to_variable = { OCF_number_military_personnel = 1 }
    
    multiply_variable = { OCF_spending_funds_variable = OCF_spending_funds_variable_has_war }
    multiply_variable = { OCF_spending_funds_variable = OCF_number_military_personnel }#减去军队花费
  }
  #军费修正
  set_variable = { OCF_obtain_banknotes_variable_markup = 0 }
  add_to_variable = { OCF_obtain_banknotes_variable_markup = adjusting_military_spending_int }

  set_variable = { military_expenditure_impact = adjusting_military_spending_int }
  add_to_variable = { military_expenditure_impact = -30 }#默认值
  multiply_variable = { military_expenditure_impact = 0.3 }#加成数值
  set_variable = { adjusting_military_spending_int_100 = military_expenditure_impact }#加成数值
  multiply_variable = { adjusting_military_spending_int_100 = 0.01 }#加成数值

  #教育修正
  add_to_variable = { OCF_obtain_banknotes_variable_markup = adjusting_education_expenses_int }

  set_variable = { education_revision = adjusting_education_expenses_int }
  multiply_variable = { education_revision = 0.1 }
  add_to_variable = { education_revision = -10 }#默认值
  set_variable = { education_revision_100 = education_revision }#加成数值
  multiply_variable = { education_revision_100 = 0.01 }#加成数值

  #医疗修正
  add_to_variable = { OCF_obtain_banknotes_variable_markup = adjusting_medical_treatment_int }

  set_variable = { medical_correction = adjusting_medical_treatment_int }
  add_to_variable = { medical_correction = -30 }#默认值
  multiply_variable = { medical_correction = 0.35 }#加成数值
  set_variable = { medical_correction_100 = medical_correction }#加成数值
  multiply_variable = { medical_correction_100 = 0.01 }#加成数值



  multiply_variable = { OCF_obtain_banknotes_variable_markup = 0.1 }
  add_to_variable = { OCF_obtain_banknotes_variable_markup = 1 }#默认值
  multiply_variable = { OCF_spending_funds_variable = OCF_obtain_banknotes_variable_markup }
  subtract_from_variable = { OCF_bill = OCF_spending_funds_variable }#每月花费
}
